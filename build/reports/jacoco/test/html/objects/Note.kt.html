package objects

data class Note(
    val id: Int,
    val title: String,
    val text: String,
    val isDeleted: Boolean = false
) {
    override fun toString(): String {
        val commentsSB = StringBuilder()
        for (comment in NoteService.getComments(id)) {
            commentsSB.append("\n$comment")
        }
        return """
           |***********
           |Note id: $id
           |Title: $title
           |Text: $text
           |Comments: $commentsSB
        """.trimMargin()
    }
}
